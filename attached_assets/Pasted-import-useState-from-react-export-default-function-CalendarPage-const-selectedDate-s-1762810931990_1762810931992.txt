          onClick={() => {
            setSelectedDate(10);
            setView("day");
          }}

        >
          Today
        </button>
      </header>

      {/* Month Navigation */}

          </div>

              <div key={d}>{d}</div>
            ))}
          </div>

          <div className="grid grid-cols-7 gap-2 p-3">
            {days.map((day) => (
              <button

                onClick={() => {
                  setSelectedDate(day);
                  setView("day");
                }}
                className={`aspect-square rounded-xl flex items-center justify-center transition text-sm font-medium ${
                  selectedDate === day
                    ? "bg-blue-600 font-bold shadow-lg shadow-blue-600/30"
                    : "hover:bg-blue-600/20"
                }`}
              >
                {day}
              </button>
            ))}
          </div>
        </>
      )}

      {/* Day View */}
      {view === "day" && (
        <div className="flex flex-col flex-1">
          <div className="flex items-center justify-between px-4 py-3 border-b border-white/10">

            </div>

        +
      </button>
    </div>
